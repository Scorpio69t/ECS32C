int main(int argc, char** argv) {
    FILE * inFile;
    FILE * outFile
    inFile = fopen(argv[1], "r");
    outFile = fopen(argv[2], "w");
    char c;
    while(1) {
        c = fgetc(inFile);
        if (c == EOF) 
            break;
        fputc(c, outFile); 
    }
    fclose(inFile);
    fclose(outFile);
}
